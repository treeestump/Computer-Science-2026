Algorithm pseudocode

INITIALIZE global variables:
    totalSlots = 120
    occupied = 0
    garageCode = "G"
    ratePerHour = 2.50
    isWeekend = False

FUNCTION generate_ticket(vehicle_plate, hoursText):
    Convert hoursText to integer as localHours
    
    IF occupied >= totalSlots THEN
        RETURN "Garage Full"
    END IF
    
    IF isWeekend is True THEN
        SET localRate = 3.00
    ELSE
        SET localRate = ratePerHour (2.50)
    END IF
    
    CALCULATE localCost = localHours * localRate
    
    BUILD ticket string: garageCode + "-" + vehicle_plate + "-" + occupied
    
    OPEN "tickets.txt" in append mode
    WRITE newline + ticket + ", " + localCost to file
    CLOSE file
    
    INCREMENT occupied by 1
    
    RETURN ticket + " costs " + localCost
END FUNCTION

MAIN EXECUTION:
    PRINT generate_ticket("AB12CD", "2")
    PRINT generate_ticket("ZZ99YY", "4")
    SET isWeekend = True
    PRINT generate_ticket("K1NG", "3")
    WAIT for user input
